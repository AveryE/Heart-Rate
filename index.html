// Declare all variables
var userAge;
var restingHeartRate;
var maximumHeartRate;

function getAge() { // Get age from user userAge = document.getElementById("age").value;

// Convert the entered age to number form
userAge = Number(userAge);

// send alert if not a number
if (isNaN(userAge)) {
  alert("Age must be entered in number form");
}

// Send alert if a negative number if (userAge < 0) { alert("Age must be a positive number"); } }

function calculateMaximumHeartRate() { // Get a Celsius temperature from user maximumHeartRate = 220 - userAge; }

function getRestingHeartRate() { // Get resting heart rate from user restingHeartRate = document.getElementById("resting").value;

// Convert to number form
restingHeartRate = Number(restingHeartRate);

// send alert if not a number
if (isNaN(restingHeartRate)) {
  alert("Temperature must be entered in number form");
}

// Send alert if a negative number
if (restingHeartRate< 0) {
  alert("Heart rate must be a positive number");
}

}

function calculateHeartRateReserve() { heartRateReserve = maximumHeartRate - restingHeartRate; }

function calculateTargetHeartRate() { targetFifty = (heartRateReserve * .5) + restingHeartRate; targetFiftyFive = (heartRateReserve * .55) + restingHeartRate; targetSixty = (heartRateReserve * .6) + restingHeartRate; targetSixtyFive = (heartRateReserve * .65) + restingHeartRate; targetSeventy = (heartRateReserve * .7) + restingHeartRate; targetSeventyFive = (heartRateReserve * .75) + restingHeartRate; targetEighty = (heartRateReserve * .8) + restingHeartRate; targetEightyFive = (heartRateReserve * .85) + restingHeartRate;

}

function outputTargetHeartRateTable() {

// Create reference on web page for output
outputToPage = document.getElementById("target");

// Output results to page, or notify of error
if (isNaN(userAge)) {      
  outputToPage.innerHTML = "Age not entered in number form"
          + " so conversion impossible.";
} else if (isNaN(restingHeartRate)) {      
  outputToPage.innerHTML = "Heart rate not entered in number form"
          + " so conversion impossible.";
} else {
  outputToPage.innerHTML = "You entered your age as " 
          + userAge 
          + " and your resting heart rate as "
          + restingHeartRate
          + ".<br /><br />"
          + "Your target heart rate table: "
          + "<br /><table><tr><td><b>Percentage of Reserve</b></td><td></td>"
          + "<td><b>Target Heart Rate</b></td></tr>"
          + "<tr><td>50%</td><td></td><td>"
          + targetFifty.toFixed(0)
          + "</td></tr>"
          + "<tr><td>55%</td><td></td><td>"
          + targetFiftyFive.toFixed(0)
          + "</td></tr>"
          + "<tr><td>60%</td><td></td><td>"
          + targetSixty.toFixed(0)
          + "</td></tr>"
          + "<tr><td>65%</td><td></td><td>"
          + targetSixtyFive.toFixed(0)
          + "</td></tr>"
          + "<tr><td>70%</td><td></td><td>"
          + targetSeventy.toFixed(0)
          + "</td></tr>"
          + "<tr><td>75%</td><td></td><td>"
          + targetSeventyFive.toFixed(0)
          +"</td></tr>"
          + "<tr><td>80%</td><td></td><td>"
          + targetEighty.toFixed(0)
          + "</td></tr>"
          + "<tr><td>85%</td><td></td><td>"
          + targetEightyFive.toFixed(0)
          +"</td></tr></table>";
}

} function heartRate() { getAge(); calculateMaximumHeartRate(); getRestingHeartRate(); calculateHeartRateReserve(); calculateTargetHeartRate(); outputTargetHeartRateTable();

return false;

}
